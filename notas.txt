Ejemplo 3: Función
-En JavaScript
function saludo(nombre) {
    console.log("Hola, " + nombre);
}
saludo("axelito");

-En Pyton:
def saludo(nombre):
    print("Hola, " + nombre)

saludo("axelito")

Ejemplo 4: Condicional (if-else)
-En JavaScript;
let edad = 18;
if (edad >= 18) {
    console.log("Eres adulto");
} else {
    console.log("Eres menor de edad");
}

-En Pyton;
edad = 18
if edad >= 18:
    print("Eres adulto")
else:
    print("Eres menor de edad")

Ejemplo 5: Bucle for nota: si hace la convercion bien de js a python solo que de js a pyton me manda un let de mas 
-En javascript;
for (let i = 0; i < 5; i++) {
    console.log(i);
}

-Conversión a Python
for i in range(0, 5):
    print(i)

Ejemplo 7: Bucle while;
-conversion de js:
let i = 1;
while (i <= 5) {
    console.log(i);
    i++;
}

-conversion a python:
i = 1
while i <= 5:
    print(i)
    i += 1
